FROM DEPT D
FULL JOIN EMP E
ON D.DEPTNO= E.deptno
WHERE LOC LIKE 'CHICAGO' 
5.
SELECT ENAME, D.LOC
FROM DEPT D
FULL JOIN EMP E

6.

FROM DEPT D

7.
SELECT COUNT(DISTINCT E.ENAME) AS 'TOTEMP',
	       COUNT(DISTINCT D.DEPTNO) AS 'TOTDEPT'
		   FROM EMP E, DEPT D;

     Else 'NON SALES'
     END AS 'CHECK'
      from DEPT

Select SAL,
      CASE
     WHEN SAL > 1000 THEN SAL
     Else 1000
     END AS 'DEFAULT SALARY'
      from EMP
